785  subMenuAlignment = "ALIGN_ABOVE_PARENT";
786  break;

790  for (const auto &element : m_uiElements)
791  {
792  // get all uiElements that have a _sub suffix in their MenuGroupID

794  {
795  std::string parentGroupName;
796  parentGroupName = element->getUiElementData().buildMenuID;
797  utils::strings::removeSubString(parentGroupName, subMenuSuffix);
798 
799  // set Layout parameters
800  auto &parentLayout = m_layoutGroups[parentGroupName].layout;
801  parentLayout.alignment = subMenuAlignment;
802  parentLayout.layoutType = layoutType;
803  parentLayout.padding = 8;
804  parentLayout.paddingToParent = 16;
805  }
